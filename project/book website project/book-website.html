Responsive book website with Html CSS and JavaScript Video

If you want to see the video on how to make a responsive book website here is the video below if you like this video then maybe go to my youtube channel and give me a subscribe.

Create a responsive book website steps

If you want to make a website to sell books here are the steps

1. Create a folder. You can name this folder whatever you want, and inside this folder, create the mentioned files.

2. Create an index.html file. The file name must be index and its extension .html

3. Create a style.css file. The file name must be style and its extension .css

4. Create an script.js file. The file name must be script and its extension .js

4. Create an image folder. In this folder save the photo you want to use for your website.

To start, add the following HTML codes to your index.html file to create a button that changes your background. This code includes a div and a button.

// Menu variables
let menu = document.getElementById('menu');
let menuOpenBtn = document.getElementById('openMenu');
let menuCloseBtn = document.getElementById('closeMenu');

// Search var
let search = document.getElementById('search');
let openSearchBtn = document.getElementById('open-search');
let closeSearchBtn = document.getElementById('close-search-btn');

// To top variables
let toTop  = document.getElementById('to-top');

// To top event
window.addEventListener('scroll', ()=>{
    if (window.pageYOffset > 500){
      toTop.classList.add('active-to-top');
    } else{
      toTop.classList.remove('active-to-top');
    }
});

// Menu function
function opemMenu() {
    menu.style.height = '100%';
    menu.style.display = 'block';
}

function closeMenu() {
    menu.style.height = '0%';
    menu.style.display = 'none';
}

// Search function
function openSearch() {
    search.style.width = '100%';
}

function closeSearch() {
    search.style.width = '0';
}

// Menu event
menuOpenBtn.addEventListener('click', opemMenu);

menuCloseBtn.addEventListener('click', closeMenu);

// Search event
openSearchBtn.addEventListener('click', openSearch);

closeSearchBtn.addEventListener('click', closeSearch);
